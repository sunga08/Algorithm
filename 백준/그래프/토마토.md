## 소스코드
```c++
#include <iostream>
#include <queue>

using namespace std;
int n, m;
int box[1001][1001];
int dx[4] = { -1,0,1,0 };
int dy[4] = { 0,1,0,-1 };
queue<pair<int,int>> q;
void bfs() {
    
    while (!q.empty()) {
        pair<int,int> cur = q.front();
        q.pop();
        for (int i = 0; i < 4; i++) {
            int nx = cur.first + dx[i];
            int ny = cur.second + dy[i];
            if (nx >= 0 && nx < m && ny >= 0 && ny < n && box[nx][ny] == 0) {
                box[nx][ny] = box[cur.first][cur.second] + 1;
                q.push({ nx,ny });
            }
        }
    }

}

int main()
{
    cin >> n >> m;
    for (int i = 0; i < m; i++) {
        for (int j = 0; j < n; j++) {
            scanf("%d", &box[i][j]);
        }
    }

    for (int i = 0; i < m; i++) {
        for (int j = 0; j < n; j++) {
            if (box[i][j] == 1 ) {
                q.push({ i,j });
            }
        }
    }

    bfs();

    int max = 0;
    for (int i = 0; i < m; i++) {
        for (int j = 0; j < n; j++) {
            if (box[i][j] == 0) {
                printf("-1");
                return 0;
            }
            if (max < box[i][j]) max = box[i][j];
        }
    }

    printf("%d", max - 1);
    return 0;
}
```

## 참고
