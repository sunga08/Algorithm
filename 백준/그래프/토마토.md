#include <iostream>
#include <queue>

using namespace std;
int n, m;
int box[1001][1001];
int dx[4] = { -1,0,1,0 };
int dy[4] = { 0,1,0,-1 };
queue<pair<int,int>> q;
void bfs() {
    
    while (!q.empty()) {
        pair<int,int> cur = q.front();
        q.pop();
        for (int i = 0; i < 4; i++) {
            int nx = cur.first + dx[i];
            int ny = cur.second + dy[i];
            if (nx >= 0 && nx < m && ny >= 0 && ny < n && box[nx][ny] == 0) {
                box[nx][ny] = box[cur.first][cur.second] + 1;
                q.push({ nx,ny });
            }
        }
    }

}

}
