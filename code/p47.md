## 풀이1
```c++
#include <iostream>

int main()
{
    //freopen("input.txt", "rt", stdin);
    int n, cnt = 0;
    int arr[52][52];
    scanf("%d", &n);
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= n; j++) {
            scanf("%d", &arr[i][j]);
        }
    }

    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= n; j++) {
            int tmp = arr[i][j];
            if (tmp > arr[i - 1][j] && tmp > arr[i + 1][j] && tmp > arr[i][j - 1] && tmp > arr[i][j + 1]) {
                cnt++;
            }
        }
    }

    printf("%d", cnt);

    return 0;
}
```
## 풀이2
- 방향 배열 dx,dy 사용
```c++
#include <iostream>

int arr[52][52];
int dx[4] = { -1,1,0,0 }; 
int dy[4] = { 0,0,-1,1 };

int main()
{
    //freopen("input.txt", "rt", stdin);
    int n, cnt = 0, flag;
    
    scanf("%d", &n);
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= n; j++) {
            scanf("%d", &arr[i][j]);
        }
    }

    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= n; j++) {
            flag = 0;
            for (int k = 0; k < 4; k++) {
                if (arr[i][j] <= arr[i + dx[k]][j + dy[k]]) { //나 자신보다 큰 것이 발견되면 멈추기
                    flag = 1;
                    break;
                }
            }
            if (flag == 0) cnt++;
        }
    }

    printf("%d", cnt);

    return 0;
}
```
